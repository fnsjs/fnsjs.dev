max

Computes the maximum value of array. If array is empty or falsey, undefined is returned.

import max from 'fns.js/max'

max([2, 12, 9, 4]);
// => 12

max([]);
// => undefined

Usage

ES 2015

import max from 'fns.js/max'

ESM

import { max } from 'fns.js'

CommonJS

var max = require('fns.js/max')

Params

max(array)
